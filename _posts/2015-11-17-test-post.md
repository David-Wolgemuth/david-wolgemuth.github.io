#Python Is Beautiful, Is it Not?

{% highlight python %}
class Animal:
    def __init__(self, kind):
        self.kind = kind
    def return_kind(self):
        """Return animal kind"""
        return self.kind
class Dog(Animal):
    def __init__(self, name):
        Animal.__init__(self, 'Dog')
        self.name = name
    def bark(self):
        print("I am a {} named {}".format(self.return_kind(), self.name))
joe = Dog("Joe")
joe.bark()
{% endhighlight %}
